735  void VertexTriangluation(std::vector<unsigned int>& oIndices, const std::vector<Vertex>& iVerts) {
736  // If there are 2 or less verts,
737  // no triangle can be created,
738  // so exit
739  if (iVerts.size() < 3) {
740  return;
741  }
742  // If it is a triangle no need to calculate it
743  if (iVerts.size() == 3) {
744  oIndices.push_back(0);
745  oIndices.push_back(1);
746  oIndices.push_back(2);
747  return;
748  }
749 
750  // Create a list of vertices
751  std::vector<Vertex> tVerts = iVerts;
752 
753  while (true) {
754  // For every vertex
755  for (int i = 0; i < int(tVerts.size()); i++) {
756  // pPrev = the previous vertex in the list
757  Vertex pPrev;
758  if (i == 0) {
759  pPrev = tVerts[tVerts.size() - 1];
760  }
761  else {
762  pPrev = tVerts[i - 1];
763  }
764 
765  // pCur = the current vertex;
766  Vertex pCur = tVerts[i];
767 
768  // pNext = the next vertex in the list
769  Vertex pNext;
770  if (i == tVerts.size() - 1) {
771  pNext = tVerts[0];
772  }
773  else {
774  pNext = tVerts[i + 1];
775  }
776 
777  // Check to see if there are only 3 verts left
778  // if so this is the last triangle
779  if (tVerts.size() == 3) {
780  // Create a triangle from pCur, pPrev, pNext
781  for (int j = 0; j < int(tVerts.size()); j++) {
782  if (iVerts[j].Position == pCur.Position)
783  oIndices.push_back(j);
784  if (iVerts[j].Position == pPrev.Position)
785  oIndices.push_back(j);
786  if (iVerts[j].Position == pNext.Position)
787  oIndices.push_back(j);
788  }
789 
790  tVerts.clear();
791  break;
792  }
793  if (tVerts.size() == 4) {
794  // Create a triangle from pCur, pPrev, pNext
795  for (int j = 0; j < int(iVerts.size()); j++) {
796  if (iVerts[j].Position == pCur.Position)
797  oIndices.push_back(j);
798  if (iVerts[j].Position == pPrev.Position)
799  oIndices.push_back(j);
800  if (iVerts[j].Position == pNext.Position)
801  oIndices.push_back(j);
802  }
803 
804  Vector3 tempVec;
805  for (int j = 0; j < int(tVerts.size()); j++) {
806  if (tVerts[j].Position != pCur.Position && tVerts[j].Position != pPrev.Position &&
807  tVerts[j].Position != pNext.Position) {
808  tempVec = tVerts[j].Position;
809  break;
810  }
811  }
812 
813  // Create a triangle from pCur, pPrev, pNext
814  for (int j = 0; j < int(iVerts.size()); j++) {
815  if (iVerts[j].Position == pPrev.Position)
816  oIndices.push_back(j);
817  if (iVerts[j].Position == pNext.Position)
818  oIndices.push_back(j);
819  if (iVerts[j].Position == tempVec)
820  oIndices.push_back(j);
821  }
822 
823  tVerts.clear();
824  break;
825  }
826 
827  // If Vertex is not an interior vertex
828  float angle = math::AngleBetweenV3(pPrev.Position - pCur.Position, pNext.Position - pCur.Position) *
829  (180 / 3.14159265359);
830  if (angle <= 0 && angle >= 180)
831  continue;
832 
833  // If any vertices are within this triangle
834  bool inTri = false;
835  for (int j = 0; j < int(iVerts.size()); j++) {
836  if (algorithm::inTriangle(iVerts[j].Position, pPrev.Position, pCur.Position, pNext.Position) &&
837  iVerts[j].Position != pPrev.Position && iVerts[j].Position != pCur.Position &&
838  iVerts[j].Position != pNext.Position) {
839  inTri = true;
840  break;
841  }
842  }
843  if (inTri)
844  continue;
845 
846  // Create a triangle from pCur, pPrev, pNext
847  for (int j = 0; j < int(iVerts.size()); j++) {
848  if (iVerts[j].Position == pCur.Position)
849  oIndices.push_back(j);
850  if (iVerts[j].Position == pPrev.Position)
851  oIndices.push_back(j);
852  if (iVerts[j].Position == pNext.Position)
853  oIndices.push_back(j);
854  }
855 
856  // Delete pCur from the list
857  for (int j = 0; j < int(tVerts.size()); j++) {
858  if (tVerts[j].Position == pCur.Position) {
859  tVerts.erase(tVerts.begin() + j);
860  break;
861  }
862  }
863 
864  // reset i to the start
865  // -1 since loop will add 1 to it
866  i = -1;
867  }
868 
869  // if no triangles were created
870  if (oIndices.size() == 0)
871  break;
872 
873  // if no more vertices
874  if (tVerts.size() == 0)
875  break;
876  }
877  }
