259  template<typename CoordAccessor>
260  typename Miniball<CoordAccessor>::NT Miniball<CoordAccessor>::relative_error(NT& subopt) const {
261  NT e, max_e = nt0;
262  // compute maximum absolute excess of support points
263  for (SupportPointIterator it = support_points_begin(); it != support_points_end(); ++it) {
264  e = excess(*it);
265  if (e < nt0)
266  e = -e;
267  if (e > max_e) {
268  max_e = e;
269  }
270  }
271  // compute maximum excess of any point
272  for (Pit i = points_begin; i != points_end; ++i)
273  if ((e = excess(i)) > max_e)
274  max_e = e;
275 
276  subopt = suboptimality();
277  assert(current_sqr_r > nt0 || max_e == nt0);
278  return (current_sqr_r == nt0 ? nt0 : max_e / current_sqr_r);
279  }

292  template<typename CoordAccessor>
293  void Miniball<CoordAccessor>::mtf_mb(Sit n) {
294  // Algorithm 1: mtf_mb (L_{n-1}, B), where L_{n-1} = [L.begin, n)
295  // B: the set of forced points, defining the current ball
296  // S: the superset of support points computed by the algorithm
297  // --------------------------------------------------------------
298  // from B. Gaertner, Fast and Robust Smallest Enclosing Balls, ESA 1999,
299  // http://www.inf.ethz.ch/personal/gaertner/texts/own_work/esa99_final.pdf
300 
301  // PRE: B = S
302  assert(fsize == ssize);
303 
304  support_end = L.begin();
305  if ((fsize) == d + 1)
306  return;
307 
308  // incremental construction
309  for (Sit i = L.begin(); i != n;) {
310  // INV: (support_end - L.begin() == |S|-|B|)
311  assert(std::distance(L.begin(), support_end) == ssize - fsize);
312 
313  Sit j = i++;
314  if (excess(*j) > nt0)
315  if (push(*j)) { // B := B + p_i
316  mtf_mb(j); // mtf_mb (L_{i-1}, B + p_i)
317  pop(); // B := B - p_i
318  mtf_move_to_front(j);
319  }
320  }
321  // POST: the range [L.begin(), support_end) stores the set S\B
322  }

331  template<typename CoordAccessor>
332  void Miniball<CoordAccessor>::pivot_mb(Pit n) {
333  // Algorithm 2: pivot_mb (L_{n-1}), where L_{n-1} = [L.begin, n)
334  // --------------------------------------------------------------
335  // from B. Gaertner, Fast and Robust Smallest Enclosing Balls, ESA 1999,
336  // http://www.inf.ethz.ch/personal/gaertner/texts/own_work/esa99_final.pdf
337  NT old_sqr_r;
338  const NT* c;
339  Pit pivot, k;
340  NT e, max_e, sqr_r;
341  Cit p;
342  do {
343  old_sqr_r = current_sqr_r;
344  sqr_r = current_sqr_r;
345 
346  pivot = points_begin;
347  max_e = nt0;
348  for (k = points_begin; k != n; ++k) {
349  p = coord_accessor(k);
350  e = -sqr_r;
351  c = current_c;
352  for (int j = 0; j < d; ++j)
353  e += mb_sqr<NT>(*p++ - *c++);
354  if (e > max_e) {
355  max_e = e;
356  pivot = k;
357  }
358  }
359 
360  if (max_e > nt0) {
361  // check if the pivot is already contained in the support set
362  if (std::find(L.begin(), support_end, pivot) == support_end) {
363  assert(fsize == 0);
364  if (push(pivot)) {
365  mtf_mb(support_end);
366  pop();
367  pivot_move_to_front(pivot);
368  }
369  }
370  }
371  } while (old_sqr_r < current_sqr_r);
372  }

397  template<typename CoordAccessor>
398  bool Miniball<CoordAccessor>::push(Pit pit) {
399  int i;
400  NT eps = mb_sqr<NT>(std::numeric_limits<NT>::epsilon());
401 
402  Cit cit = coord_accessor(pit);
403  Cit p = cit;
404 
405  if (fsize == 0) {
406  for (i = 0; i < d; ++i)
407  q0[i] = *p++;
408  for (i = 0; i < d; ++i)
409  c[0][i] = q0[i];
410  sqr_r[0] = nt0;
411  }
412  else {
413  // set v_fsize to Q_fsize
414  for (i = 0; i < d; ++i)
415  // v[fsize][i] = p[i]-q0[i];
416  v[fsize][i] = *p++ - q0[i];
417 
418  int j;
419 
420  // compute the a_{fsize,i}, i< fsize
421  for (i = 1; i < fsize; ++i) {
422  a[fsize][i] = nt0;
423  for (j = 0; j < d; ++j)
424  a[fsize][i] += v[i][j] * v[fsize][j];
425  a[fsize][i] *= (2 / z[i]);
426  }
427 
428  // update v_fsize to Q_fsize-\bar{Q}_fsize
429  for (i = 1; i < fsize; ++i) {
430  for (j = 0; j < d; ++j)
431  v[fsize][j] -= a[fsize][i] * v[i][j];
432  }
433 
434  // compute z_fsize
435  z[fsize] = nt0;
436  for (j = 0; j < d; ++j)
437  z[fsize] += mb_sqr<NT>(v[fsize][j]);
438  z[fsize] *= 2;
439 
440  // reject push if z_fsize too small
441  if (z[fsize] < eps * current_sqr_r) {
442  return false;
443  }
444 
445  // update c, sqr_r
446  p = cit;
447  NT e = -sqr_r[fsize - 1];
448  for (i = 0; i < d; ++i)
449  e += mb_sqr<NT>(*p++ - c[fsize - 1][i]);
450  f[fsize] = e / z[fsize];
451 
452  for (i = 0; i < d; ++i)
453  c[fsize][i] = c[fsize - 1][i] + f[fsize] * v[fsize][i];
454  sqr_r[fsize] = sqr_r[fsize - 1] + e * f[fsize] / 2;
455  }
456  current_c = c[fsize];
457  current_sqr_r = sqr_r[fsize];
458  ssize = ++fsize;
459  return true;
460  }

462  template<typename CoordAccessor>
463  typename Miniball<CoordAccessor>::NT Miniball<CoordAccessor>::suboptimality() const {
464  NT* l = new NT[d + 1];
465  NT min_l = nt0;
466  l[0] = NT(1);
467  for (int i = ssize - 1; i > 0; --i) {
468  l[i] = f[i];
469  for (int k = ssize - 1; k > i; --k)
470  l[i] -= a[k][i] * l[k];
471  if (l[i] < min_l)
472  min_l = l[i];
473  l[0] -= l[i];
474  }
475  if (l[0] < min_l)
476  min_l = l[0];
477  delete[] l;
478  if (min_l < nt0)
479  return -min_l;
480  return nt0;
481  }
